function bubbleSort(arr){
    var len = arr.len;
    if(len<=1){
        return arr;
    }
    for(var i = 0; i < arr.length; i++){
        for(var j = 0; j < len - 1 - i; j++){
            if(arr[j] < arr[j + 1]){
                var temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    return arr;
}
